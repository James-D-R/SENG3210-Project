affinity SMALLINT,

slot2 SMALLINT,
slot3 SMALLINT,

red SMALLINT NOT NULL,
orange SMALLINT NOT NULL,
yellow SMALLINT NOT NULL,
green SMALLINT NOT NULL,
blue SMALLINT NOT NULL,
white SMALLINT NOT NULL,
purple SMALLINT NOT NULL
);
